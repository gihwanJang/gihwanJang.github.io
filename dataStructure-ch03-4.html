이진 탐색 트리(Binary Search Tree)

이진 탐색 트리는 이진 트리의 한 종류로서 각 노드들은 왼쪽 자식과 오른쪽 자식 2개의 자식을 같습니다.
이때 노드의 왼쪽 하위 트리에는 노드의 값보다 작은 값이있는 노드 만 포함되고 노드의 오른쪽 하위 트리에는 노드의 값보다 큰 값이있는 노드 만 포함됩니다.
그러므로 값의 중복을 허용하지 않습니다.
또한 이진 탐색 트리는 모든 하위 트리는 위의 이진 탐색 트리의 조건을 만족합니다.

(코드) 

import java.util.Iterator;
import java.util.Queue;
import java.util.Stack;
import java.util.LinkedList;
import java.util.ArrayList;
import java.util.List;

class BST<E extends Comparable<E>> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private Node right=null;
		private Node left=null;
		private Node(Object value) {
			this.item=value;
		}
	}
	
	private class BSTIterator implements Iterator<E>{
		private  int cursor =0;
		@Override
		public boolean hasNext() {
			return cursor<visitedOrder.size();
		}
		@Override
		public E next() {
			return (E)visitedOrder.get(cursor++).item;
		}
	}
	
	private Node root = null;
	private int size = 0;
	private List<Node> visitedOrder;
	
	private void BFS(Node node){
		if(this.root==null)return;
		Queue<Node> queue = new LinkedList<>();
		queue.add(node);
		while(!queue.isEmpty()) {
			Node curr=queue.poll();
			this.visitedOrder.add(curr);
			if(curr.left!=null)
				queue.add(curr.left);
			if(curr.right!=null)
				queue.add(curr.right);
		}
	}

	private void preorder(Node node){
		if(this.root==null)return;
		this.visitedOrder.add(node);
		if(node.left!=null) preorder(node.left);
		if(node.right!=null) preorder(node.right);
	}
	
	private void inorder(Node node){
		if(this.root==null)return;
		if(node.left!=null) inorder(node.left);
		this.visitedOrder.add(node);
		if(node.right!=null) inorder(node.right);
	}
	
	private void postorder(Node node){
		if(this.root==null)return;
		if(node.left!=null) postorder(node.left);
		if(node.right!=null) postorder(node.right);
		this.visitedOrder.add(node);
	}
	
	private Node findNode(Node node, Object val) {
		if(((E)node.item).compareTo((E)val)==0) return node;
		else {
			Node nextNode=((E)node.item).compareTo((E)val)>0? node.left:node.right;
			return nextNode!=null? findNode(nextNode, val): node;
		}
	}
	
	private Node findNode(Node node, Object val, Stack<Node> parents) {
		if(((E)node.item).compareTo((E)val)==0) return node;
		else {
			Node nextNode=((E)node.item).compareTo((E)val)>0? node.left:node.right;
			if(nextNode!=null) {
				parents.push(node);
				return findNode(nextNode, val, parents);
			}
			return node;
		}
	}
	
	private int height(Node node) {
		if(node==null)return 0;
		int h=height(node.left)>height(node.right)?height(node.left):height(node.right);
		return 1+h;
	}
	
	private void removeChild(Node parent,Node child ) {
		Node descendant=child.left!=null? child.left: child.right;
		if(parent==null) this.root = descendant;
		else {
			if(((E)parent.item).compareTo((E)child.item)>=0)parent.left=descendant;
			else parent.right = descendant;
		}
		--this.size;
	}
	
	private void balanceTree(int lo, int hi) {
		if(lo==hi) this.add(visitedOrder.get(lo).item);
		else if(lo+1==hi) {
			this.add(visitedOrder.get(hi).item);
			this.add(visitedOrder.get(lo).item);
		}
		else {
			int mid=lo+(hi-lo)/2;
			this.add(visitedOrder.get(mid).item);
			this.balanceTree(lo,mid-1);
			this.balanceTree(mid+1,hi);
		}
	}
	
	public BST() {
		this.root=null;
		this.size=0;
	}
	
	public BST(Object... arr) {
		for(Object v:arr) this.add(v);
	}
	
	public boolean isEmpty() {
		return this.size==0?true:false;
	}
	
	public int size() {
		return this.size;
	}
	
	public int height() {
		return height(this.root)-1;
	}
	
	public void clear() {
		this.size=0;
		this.root=null;
	}
	
	public Object get(int index) {
		if(index<0||index>this.size-1)return null;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		return visitedOrder.get(index);
	}
	
	public boolean contains(Object val) {
		if(this.isEmpty())return false;
		if(findNode(this.root,val).item==val)return true;
		return false;
	}
	
	public int indexOf(Object val) {
		if(this.isEmpty())return -1;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		int lo=0;
		int hi=this.size-1;
		while(lo<=hi){
			int mid=lo+(hi-lo)/2;
			if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)==0) return mid;
			else if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)>0){
				if(mid==0)return -1;
				hi=mid-1;
			}
			else lo=mid+1;
		}
		return -1;
	}
	
	public void add(Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
		}
		else {
			Node parent=findNode(this.root, val);
			if(((E)parent.item).compareTo((E)val)==0)return;
			else if(((E)parent.item).compareTo((E)val)>0) parent.left=newNode;
			else parent.right=newNode;
		}
		++this.size;
	}
	
	public void remove(Object val) {
		if(this.isEmpty()) return;
		Stack<Node> parents=new Stack<>();
		Node delNode = findNode(this.root, val, parents);
		if(((E)delNode.item).compareTo((E)val)!=0) return;
		if(delNode.left!=null&&delNode.right!=null) {
			Node parent=delNode;
			Node prevNode=delNode.left;
			while(prevNode.right!=null) {
				parent =prevNode;
				prevNode = prevNode.right;
			}
			delNode.item=prevNode.item;
			removeChild(parent, prevNode);
		}
		else {
			Node parent=parents.isEmpty()? null:parents.lastElement();
			removeChild(parent,delNode);
		}
	}
	
	public void balanceTree() {
		this.visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		this.clear();
		balanceTree(0, visitedOrder.size()-1);
	}
	
	public BST<E> clone() throws CloneNotSupportedException{
		BST<E> cloned = new BST<E>();
		this.visitedOrder = new ArrayList<>(this.size); 
		this.preorder(this.root);
		for(Node n: visitedOrder)
			cloned.add(n.item);
		return cloned;
	}

	public Iterator<E> iterator(){
		this.visitedOrder= new ArrayList<>(this.size);
		inorder(this.root);
		return new BSTIterator();
	}
	
	@Override
	public String toString() {
		this.visitedOrder=new ArrayList<>(this.size);
		BFS(this.root);
		int childNum=0;
		String str="";
		if(this.isEmpty())return str;
		if(this.visitedOrder.get(0).left!=null)++childNum;
		if(this.visitedOrder.get(0).right!=null)++childNum;
		str+="value : "+this.visitedOrder.get(0).item+", child : "+childNum+"\n";
		for(int i=this.height(),index=1;i>0;--i) {
			int levelsChild=0;
			for(int j=0; j<childNum;++j) {
				int currNodeChild=0;
				if(this.visitedOrder.get(index).left!=null)++currNodeChild;
				if(this.visitedOrder.get(index).right!=null)++currNodeChild;
				str+="value : "+visitedOrder.get(index).item+", child : "+currNodeChild+" ";
				levelsChild+=currNodeChild;
				++index;
			}
			childNum=levelsChild;
			str+="\n";
		}
		return str;
	}
}

left : 왼쪽 자식 노드를 저장하는 변수로 기본 값이 null로 선언되어 있습니다.

이진 탐색 트리를 후위 순회하는 메소드입니다.

위의 순서로 방문하기 때문에 preorder로 생성된 visitedOrder는 주로 트리를 삭제하는 메소드에 많이 사용 됩니다.
이유는 부모노드를 삭제하기 전에 자식 노드를 삭제하는 순으로 노드를 삭제해야 하기 때문입니다.

특정 값의 노드 혹은 해당 값의 부모가 될 노드를 찾는 메소드 입니다.
노드의 값이 특정 값과 같으면 해당 노드를 반환합니다.
아닐 시 다음 확인할 노드를 선언합니다.
이때 다음 노드는 노드의 값이 특정 값보다 크면 해당 노드의 왼쪽 자식 노드가 되고 아니면 오른쪽 자식 노드가 됩니다.
최종적으로 다음 노드가 null이 아니면 재귀적으로 특정 값에 가장 가까운 말단 값을 찾아 반환합니다.

	private Node findNode(Node node, Object val, Stack<Node> parents) {

		else {

			if(nextNode!=null) {
				parents.push(node);
				return findNode(nextNode, val, parents);

			return node;
		}

특정 값의 노드를 찾고 해당 값의 부모 노드들의 리스트를 만드는 메소드 입니다.

이때 다음 노드는 노드의 값이 특정 값보다 크면 해당 노드의 왼쪽 자식 노드가 되고 아니면 오른쪽 자식 노드가 됩니다.
이때 다음 노드가 null이 아니라면 즉 찾는 노드가 아니라면 부모노드들을 저장하는 parents스텍에 push하고 찾는 노드라면 push하지 않고 반환합니다.

특정 노드와 특정 노드의 부모노드를 받아 특정 노드를 삭제하는 메소드입니다.
삭제 될 노드의 후손 노드를 선언해 줍니다.
해당 후손 노드는 만약 왼쪽 자식 노드가 있다면 왼쪽 자식 노드로 없다면 오른쪽 자식 노드로 지정해 줍니다.
만약 부모 노드가 null이라면 삭제 될 노드는 root이므로 후손 노드를 root로 지정해 줍니다.
null이 아니라면 부모노드의 값이 삭제 노드의 값 보다 크거나 같다면 자손 노드는 부모 노드의 왼쪽 자식이 되고 작다면 오른쪽 자식 노드가 됩니다.
최종적으로 노드가 삭제됬음으로 size를 감소 시켜줍니다.

(balanceTree) 

	private void balanceTree(int lo, int hi) {
		if(lo==hi) this.add(visitedOrder.get(lo).item);
		else if(lo+1==hi) {
			this.add(visitedOrder.get(hi).item);
			this.add(visitedOrder.get(lo).item);
		}
		else {
			int mid=lo+(hi-lo)/2;
			this.add(visitedOrder.get(mid).item);
			this.balanceTree(lo,mid-1);
			this.balanceTree(mid+1,hi);
		}
	}

완전 이진 트리가 되도록 트리를 재구성하는 메소드입니다.
이진 탐색 트리는 특정 값을 기준으로 작으면 좌측 크면 우측에 위치하기 때문에 완전 이진 트리를 만들기 위해서는 중앙 값을 넣고 해당 중앙 값을 기준으로 나눈 중앙 값을 반복하여 넣으면 됩니다.
inorder로 생성된 정렬 리스트 visitedOrder를 이용하여 처음 인덱스 low값과 마즈막 인덱스 hight 값을 받습니다.
이때 lo 와 hi 가 같다면 값이 하나 밖에 없음으로 해당 값을 트리에 add해 줍니다.
lo+1이 hi와 같다면 mid가 없고 값이 2개 뿐 이므로 해당 두 값들을 트리에 add해 줍니다.
위의 경우가 아니라면 lo와 hi를 더하고 나누기 2를 하여 mid 중앙 인덱스를 찾고 해당 중앙 인덱스 값을 트리에 add하고 해당 메소드에 mid를 기준으로 lo~mid-1, mid+1,hi로 나누어 해당 메소드를 재귀 호출하여 트리를 완전 이진 트리로 재구성합니다.

	}

트리를 초기화 하는 메소드입니다.
size를 0으로 root를 null로 초기화해 줍니다.

(get) 

	public Object get(int index) {
		if(index<0||index>this.size-1)return null;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		return visitedOrder.get(index);
	}

트리의 특정 인덱스 값을 받아오는 메소드입니다.
inorder를 통해 정렬된 리스트 visitedOrder를 생성 후 해당 인덱스의 값을 반환합니다.

(contains) 

	public boolean contains(Object val) {
		if(this.isEmpty())return false;
		if(findNode(this.root,val).item==val)return true;
		return false;
	}

트리에 특정 값의 노드가 존재하는지 확인하는 메소드입니다. 해당 트리가 빈 트리라면 false를 리턴합니다.
아니라면 findNode메소드에 root와 특정 값을 주고 반환된 노드의 값이 특정 값과 같으면 true를 아니면 false를 반환합니다.

(indexOf) 

	public int indexOf(Object val) {
		if(this.isEmpty())return -1;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		int lo=0;
		int hi=this.size-1;
		while(lo<=hi){
			int mid=lo+(hi-lo)/2;
			if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)==0) return mid;
			else if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)>0){
				if(mid==0)return -1;
				hi=mid-1;
			}
			else lo=mid+1;
		}
		return -1;
	}

트리에 특정 값의 노드가 위치하는 인덱스를 반환하는 메소드입니다. inorder를 통해 정렬된 리스트 visitedOrder를 생성 후 이진 탐색을 통해 해당 값이 있으면 해당 인덱스를 없다면 -1을 반환합니다.

(add) 

	public void add(Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
		}
		else {
			Node parent=findNode(this.root, val);
			if(((E)parent.item).compareTo((E)val)==0)return;
			else if(((E)parent.item).compareTo((E)val)>0) parent.left=newNode;
			else parent.right=newNode;
		}
		++this.size;
	}

트리에 새로운 값의 노드를 추가하는 메소드입니다.
주어진 값을 가지는 노드를 생성합니다.
트리가 빈트리 라면 새로 생성한 노드는 root가 됩니다.
아니라면 findNode에 root 부터 탐색하기 위해 root와 특정 값 val을 주어 부모노드를 받아옵니다.
이때 해당 값이 이미 트리에 존재하여 findNode를 통해 부모노드가 아닌 해당 값을 가진 노드가 반환 될 수 있음으로 해당 부모 노드와 입력 값을 비교하여 같다면 멈춥니다.
같지않다면 해당 부모 노드의 값과 입력 값을 비교하여 부모노드 보다 작다면 왼쪽 자식으로 크다면 오른쪽 자식으로 지정해 줍니다.
최종적으로 노드가 추가 됬음으로 size를 증가시켜줍니다.

(remove) 

	public void remove(Object val) {
		if(this.isEmpty()) return;
		Stack<Node> parents=new Stack<>();
		Node delNode = findNode(this.root, val, parents);
		if(((E)delNode.item).compareTo((E)val)!=0) return;
		if(delNode.left!=null&&delNode.right!=null) {
			Node parent=delNode;
			Node prevNode=delNode.left;
			while(prevNode.right!=null) {
				parent =prevNode;
				prevNode = prevNode.right;
			}
			delNode.item=prevNode.item;
			removeChild(parent, prevNode);
		}
		else {
			Node parent=parents.isEmpty()? null:parents.lastElement();
			removeChild(parent,delNode);
		}
	}

트리에서 특정 값을 삭제하는 메소드입니다.
삭제는 3가지 경우로 나누어 해당 값을 삭제 하였습니다.
우선 해당 값의 노드가 말단 노드일 경우 바로 해당 노드를 삭제합니다.
두번째로 해당 값의 노드의 자식 노드가 1개인 경우 해당 자식 노드를 부모 노드에 해당 노드가 있던 위치에 연결시켜줍니다.
마즈막으로 해당 값의 노드의 자식이 둘인 경우 왼쪽 부분 트리 중 값이 가장 큰 노드와 값을 교체후 교체한 말단 노드를 삭제합니다.
위의 경우에 따라 노드를 삭제를 하기위해서는 부모 노드가 필요하므로 부모 노드를 받는 parents리스트를 선언해 줍니다.
findNode메소드를 통해 parents의 리스트를 받고 삭제할 노드를 받아옵니다.
이때 삭제할 노드가 없다면 해당 메소드는 종료 됩니다.
만약 삭제 노드의 자식이 두개라면 왼쪽 자식 노드의 자식들 중 가장 우측 하단에 있는 노드 즉 좌측 서브 트리의 가장큰 값의 노드를 찾고 해당 노드와 삭제할 노드의 값을 바꿉니다.
삭제노드를 바꾼 노드로 바꾸고 removeChild메소드를 통해 삭제해 줍니다.
자식이 2개가 아닌 경우 removechild 메소드를 통해 해당 노드를 삭제해 줍니다.

(balanceTree) 

	public void balanceTree() {
		this.visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		this.clear();
		balanceTree(0, visitedOrder.size()-1);
	}

트리를 완전 이진 트리로 재구성하는 메소드 입니다.
visitedOrder를 inorder메소드를 통해 정렬된 리스트로 생성합니다.
해당 트리를 clear메소드를 통해 초기화합니다.
balanceTree메소드를 통해 트리를 재구성 해줍니다.
