자료구조 Ch.02 : 리스트

6-2. 비정렬 단일 연결 리스트(UnsortedSingleLinkedList)with tail

앞의 비정렬 단일 연결 리스트에서는 해당 리스트가 head 만을 가지고 있어 tail을 추가해 보았습니다.
tail을 추가하므로서 pushBack과 peekBack을 O(1)의 시간복잡도로 처리할 수 있게 됩니다.
아래에서 수정된 부분만 추가설명을 하겠습니다.

(코드) 

import java.util.Iterator;
import java.util.Objects;

class SingleUnsortedLinkedList<E> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private Node next=null;
		private Node(Object value) {
			this.item=value;
		}
		public Node clone() throws CloneNotSupportedException{
			Node cloned = (Node)super.clone();
			return cloned;
		}
	}
	private Node head = null;
	private Node tail = null;
	private int size = 0;
	
	private class LinkedListIterator implements Iterator<E>{
		private  int cursor =0;
		private Node curr=head;
		@Override
		public boolean hasNext() {
			return cursor<size;
		}
		@Override
		public E next() {
			Object val=curr.item;
			++cursor;
			curr=curr.next;
			return (E)val;
		}
	}
	
	@Override
	public String toString() {
		String str="[";
		Node curr=head;
		for(int i=0;i<size;++i) {
			str+=curr.item;
			curr=curr.next;
			if(i!=this.size-1) {
				str+=", ";
			}
		}
		return str+"]";
	}
	
	public SingleUnsortedLinkedList(){
		this.head = null;
		this.tail = null;
		this.size = 0;
	}
	
	public SingleUnsortedLinkedList(Object... arr){
		for(int i=arr.length-1;i>=0;--i) {
			this.pushFront(arr[i]);
		}
	}
	
	public boolean isEmpty() {
		if(this.size==0)return true;
		return false;
	}
	
	public int size() {
		return this.size;
	}
	
	public void clear() {
		this.size=0;
		this.head=null;
		this.tail=null;
	}
	
	public int indexOf(Object o) {
		if(this.isEmpty())return -1;
		Node curr=head;
		for(int i=0;i<size;++i) {
			if(curr.item==o)return i;
			curr=curr.next;
		}
		return -1;
	}
	
	public boolean contains(Object o) {
		if(this.isEmpty())return false;
		return this.indexOf(o)!=-1?true:false;
	}
	
	public Object get(int index) {
		Objects.checkIndex(index, this.size-1);
		Node curr=this.head;
		for(int i=0;i<index;++i) curr=curr.next;
		return curr.item;
	}
	
	public void pushFront(Object val) {
		Node newNode = new Node(val);
		newNode.next=this.head;
		if(isEmpty())this.head=this.tail=newNode;
		else this.head=newNode;
		++this.size;
	}
	
	public void popFront() {
		if(this.isEmpty())return;
		this.head=head.next;
		--this.size;
		if(this.isEmpty())this.tail=null;
	}

	public Object peekFront() {
		return head.item;
	}
	
	public void pushBack(Object val) {
		Node newNode = new Node(val);
		if(isEmpty())head=tail=newNode;
		else{
			Node curr=this.tail;
			curr.next=newNode;
			this.tail=newNode;
		}
		++this.size;
	}
	
	public void popBack() {
		if(this.isEmpty())return;
		Node curr=this.head;
		for(int i=2; i<size; ++i) curr=curr.next;
		curr.next=null;
		this.tail=curr;
		--this.size;
	}
	
	public Object peekBack() {
		return this.tail.item;
	}
	
	public void remove(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr.item!=val) {
			curr=curr.next;
			prev=prev.next;
		}
		if(curr==this.head)this.head=this.head.next;
		if(curr==this.tail)this.tail=prev;
		else prev.next=curr.next;
		--this.size;
	}
	
	public void removeAll(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr!=null) {
			if(curr.item!=val) {
				curr=curr.next;
				prev=prev.next;
			}
			else {
				if(curr==this.head)this.head=this.head.next;
				if(curr==this.tail)this.tail=prev;
				else prev.next=curr.next;
				curr=curr.next;
				--this.size;
			}
		}
	}
	
	public Iterator<E> iterator(){
		return new LinkedListIterator();
	}
	
	public SingleUnsortedLinkedList<E> clone() throws CloneNotSupportedException{
		SingleUnsortedLinkedList<E> cloned = (SingleUnsortedLinkedList<E>)super.clone();
		Node curr=cloned.head;
		for(E i:this) {
			Node newNode = new Node(i);
			curr=newNode;
			curr=curr.next;
		}
		return cloned;
	}
}

(맴버변수) 

private Node head = null;
private Node tail = null;
private int size = 0;

맴버변수는 해당 연결 리스트의 처음과 크기를 나타내는 변수들을 사용하였습니다.
head : 연결 리스트의 처음을 나타내는 변수입니다.
tail : 연결 리스트의 끝을 나타내는 변수입니다.
size : 크기 정보 즉 노드의 갯수를 저장하는 변수 입니다.

(생성자) 

	public SingleUnsortedLinkedList(){
		this.head = null;
		this.tail = null;
		this.size = 0;
	}
	
	public SingleUnsortedLinkedList(Object... arr){
		for(int i=arr.length-1;i>=0;--i) {
			this.pushFront(arr[i]);
		}
	}

연결 리스트의 생성자입니다. SingleUnsortedLinkedArrayList() : 인자를 받지 않고 생성하는 경우 head와 tail는 null 로 size는 0으로 초기화 하여 생성합니다.
SingleUnsortedLinkedArrayList(Object... arr) : 값을 인자로 받아 생성하는 경우 해당 값끝에서 부터 연결리스트 앞에서 부터 차례대로 삽입하여 생성합니다.
값을 앞에서 부터 삽입할 때에는 pushFront 메소드를 연속적으로 호출하여 삽입하게 됩니다.

(clear) 

	public void clear() {
		this.size=0;
		this.head=null;
		this.tail=null;
	}

리스트를 초기화 해주는 메소드입니다.
head와 tail을 null로 size를 0으로 바꾸어줍니다.

(pushFront) 

	public void pushFront(Object val) {
		Node newNode = new Node(val);
		newNode.next=this.head;
		if(isEmpty())this.head=this.tail=newNode;
		else this.head=newNode;
		++this.size;
	}

리스트의 맨앞에 새로운 값을 추가하는 메소드입니다.
인자로 받은 값을 저장한 노드를 생성후 해당 노드를 next를 head로 연결해 준후 head를 생성한 새노드로 지정하고 size를 증가시킵니다.
이때 비어있는 리스트에 처음 노드를 생성할 시 해당 노드는 리스트의 head와 tail이 됩니다.

(popFront) 

	public void popFront() {
		if(this.isEmpty())return;
		this.head=head.next;
		--this.size;
		if(this.isEmpty())this.tail=null;
	}

리스트의 맨앞의 값을 삭제하는 메소드입니다.
head를 head의 next로 바꾼뒤 size를 감소시킵니다.
이때 리스트가 빈리스트가 되면 tail을 null로 바꿔줍니다.

(pushBack) 

	public void pushBack(Object val) {
		Node newNode = new Node(val);
		if(isEmpty())head=tail=newNode;
		else{
			Node curr=this.tail;
			curr.next=newNode;
			this.tail=newNode;
		}
		++this.size;
	}

리스트의 맨뒤에 새로운 값을 추가하는 메소드입니다.
만약 리스트가 비여있다면 새 값을 가진 노드를 생성후 head와 tail로 지정후 size를 증가 시키고 비어있지 않다면 tail의 다음에 새 노드를 연결하고 새노드를 tail로 바꾼후 size를 증가시킵니다.

(popBack) 

	public void popBack() {
		if(this.isEmpty())return;
		Node curr=this.head;
		for(int i=2; i<size; ++i) curr=curr.next;
		curr.next=null;
		this.tail=curr;
		--this.size;
	}

리스트의 맨뒤의 값을 삭제하는 메소드입니다.
tail의 전 노드까지 선형검색하여 해당 노드의 next를 null로 바꾼후 tail을 해당노드로 바꾸고 size를 감소시킵니다.

(peekBack) 

	public Object peekBack() {
		return this.tail.item;
	}

리스트의 맨뒤의 값을 반환하는 메소드입니다.
tail의 값을 반환해 줍니다.

(remove) 

	public void remove(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr.item!=val) {
			curr=curr.next;
			prev=prev.next;
		}
		if(curr==this.head)this.head=this.head.next;
		if(curr==this.tail)this.tail=prev;
		else prev.next=curr.next;
		--this.size;
	}

리스트의 특정 객체 맨앞의 하나를 삭제하는 메소드입니다.
indexOf 메소드를 이용해 리스트가 비거나 해당 객체가 리스트에 없으면 return합니다
객체가 리스트에 있으면 처음 노드와 해당노드의 전 노드를 선형적으로 해당 객체가 있는 노드까지 접근합니다.
해당 위치의 노드가 head이면 head를 head의 next로 바꾸고 tail이면 tail을 prev로 바꾸고 아니면 전 노드의 next를 객체가 있는 노드의 next로 바꾼 후 size를 감소시킵니다.

(removeAll) 

	public void removeAll(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr!=null) {
			if(curr.item!=val) {
				curr=curr.next;
				prev=prev.next;
			}
			else {
				if(curr==this.head)this.head=this.head.next;
				if(curr==this.tail)this.tail=prev;
				else prev.next=curr.next;
				curr=curr.next;
				--this.size;
			}
		}
	}

리스트의 특정 객체를 모두 삭제하는 메소드입니다.
indexOf 메소드를 이용해 리스트가 비거나 해당 객체가 리스트에 없으면 return합니다
객체가 리스트에 있으면 처음 노드와 해당노드의 전 노드를 선형적으로 마즈막 노드까지 접근합니다.
이때 현재의 노드의 값이 특정 객이면 해당 위치의 노드가 head이면 head를 head의 next로 바꾸고 tail이면 tail을 prev로 바꾸고 아니면 전 노드의 next를 객체가 있는 노드의 next로 바꾼 후 size를 감소시킵니다.

(정리)

contains pushFront pushBack popFront popBack remove
O(n) O(1) O(1) O(1) O(n) O(n)

위의 표와 같이 주요 메소드들의 시간 복잡도를 정리해 보았습니다.
head만 유지했을 때와 비교했을 때 pushBack과 peekBack이 O(1)에 가능한 것을 볼 수 있었습니다.
