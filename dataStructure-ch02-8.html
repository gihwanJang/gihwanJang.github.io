자료구조 Ch.02 : 리스트

9. 정렬 이중연결 리스트(SortedDoubleLinkedList)with tail

정렬 단일연결 리스트와 같이 비정렬 이중연결 리스트에서 크게 변하는 것은 없습니다.
다른 것은 모두 단일연결 리스트와 같은 방식으로 바꿔주면 되지만 add메소드에서 prev를 고려해 주어야한다는 점이 정렬 단일연결 리스트와의 차이점입니다.

(코드) 

import java.util.Iterator;
import java.util.Objects;

class SortedDoubleLinkedList<E extends Comparable<E>> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private Node prev=null;
		private Node next=null;
		private Node(Object value) {
			this.item=value;
		}
	}
	private Node head = null;
	private Node tail = null;
	private int size = 0;
	
	private class LinkedListIterator implements Iterator<E>{
		private  int cursor =0;
		private Node curr=head;
		@Override
		public boolean hasNext() {
			return cursor<size;
		}
		@Override
		public E next() {
			Object val=curr.item;
			++cursor;
			curr=curr.next;
			return (E)val;
		}
	}
	
	@Override
	public String toString() {
		String str="[";
		Node curr=head;
		for(int i=0;i<size;++i) {
			str+=curr.item;
			curr=curr.next;
			if(i!=this.size-1) {
				str+=", ";
			}
		}
		return str+"]";
	}
	
	public SortedDoubleLinkedList(){
		this.head = null;
		this.tail = null;
		this.size = 0;
	}
	
	public SortedDoubleLinkedList(Object... arr){
		for(int i=arr.length-1;i>=0;--i) {
			this.add(arr[i]);
		}
	}
	
	public boolean isEmpty() {
		if(this.size==0)return true;
		return false;
	}
	
	public int size() {
		return this.size;
	}
	
	public void clear() {
		this.size=0;
		this.head=null;
		this.tail=null;
	}
	
	public int indexOf(Object o) {
		if(this.isEmpty())return -1;
		Node curr=head;
		for(int i=0;i<size;++i) {
			if(curr.item==o)return i;
			curr=curr.next;
		}
		return -1;
	}
	
	public boolean contains(Object o) {
		if(this.isEmpty())return false;
		return this.indexOf(o)!=-1?true:false;
	}
	
	public Object get(int index) {
		Objects.checkIndex(index, this.size-1);
		Node curr=this.head;
		for(int i=0;i<index;++i) curr=curr.next;
		return curr.item;
	}
	
	public void set(int index, Object o) {
		Node curr = this.head;
		for(int i=0; i<index; ++i) curr=curr.next;
		curr.item=o;
	}
	
	public void add(Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.head=this.tail=newNode;
			++this.size;
			return;
		}
		Node curr = head;
		while(curr!=null && ((E)curr.item).compareTo((E)val)<=0) {
			curr=curr.next;
		}
		if(curr==head) {
			this.head.prev=newNode;
			newNode.next=this.head;
			this.head=newNode;
		}
		else if(curr==null) {
			this.tail.next=newNode;
			newNode.prev=this.tail;
			this.tail=newNode;
		}
		else{
			newNode.next=curr;
			newNode.prev=curr.prev;
			curr.prev.next=newNode;
			curr.prev=newNode;
		}
		++this.size;
	}

	public void popFront() {
		if(this.isEmpty())return;
		this.head=this.head.next;
		this.head.prev=null;
		--this.size;
	}

	public Object peekFront() {
		return head.item;
	}

	public void popBack() {
		if(this.isEmpty())return;
		this.tail=this.tail.prev;
		this.tail.next.prev=null;
		this.tail.next=null;
		--this.size;
	}
	
	public Object peekBack() {
		return this.tail.item;
	}
	
	public void remove(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		while(curr.item!=val) curr=curr.next;
		if(curr==this.head) this.popFront();
		else if(curr==this.tail) this.popBack();
		else{
			curr.prev.next=curr.next;
			curr.next.prev=curr.prev;
			--this.size;
		}
	}
	
	public void removeAll(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		while(((E)curr.item).compareTo((E)val)<=0) {
			if(curr.item!=val) curr=curr.next;
			else {
				if(curr==this.head)this.popFront();
				else if(curr==this.tail)this.popBack();
				else{
					curr.prev.next=curr.next;
					curr.next.prev=curr.prev;
					--this.size;
				}
				curr=curr.next;
			}
		}
	}
	
	public Iterator<E> iterator(){
		return new LinkedListIterator();
	}
	
	public SortedDoubleLinkedList<E> clone() throws CloneNotSupportedException{
		SortedDoubleLinkedList<E> cloned = new SortedDoubleLinkedList<E>();
		for(E i:this) cloned.add(i);
		return cloned;
	}
}

리스트에 compareTo 메소드를 통해 크기순으로 정렬하여 객체를 삽입해주는 메소드입니다.
단일연결에서의 prev노드를 생성하여 사용한 대신 이중연결에서는 노드 내부의 prev연결을 사용합니다.
기존대로 리스트가 비어있을 경우 새로 생성한 노드를 head와 tail로 설정해 주고 비어있지 않다면 해당 노드보다 값이 큰 노드 까지 선형 접근을 합니다.
이후 부터는 prev노드를 설정해 주어야하기 때문에 단일연결 리스트와 조금 차이가 있습니다.
우선 curr가 head일 경우 curr가 null인 경우 이외의 경우로 나눌 수 있습니다.
curr가 head 일 경우 head의 prev를 새로 생성한 노드로 설정해 주고 새로 생성한 노드의 next를 head로 설정해 주고 head를 새로 생성한 노드로 바꿉니다.
curr가 null인 경우 curr를 사용 할 수 없므로 tail의 next를 새로 생성한 노드로 설정해 주고 새로 생성한 노드의 prev를 tail로 설정해 주고 새로 생성한 노드를 tail로 설정해 줍니다.
이외의 경우는 우선 새로 생성한 노드의 next를 curr로 prev를 curr의 prev로 설정해 주고 curr의 전 노드 즉 prev의 next를 새로 생성한 노드로 curr의 prev를 새로 생성한 노드로 설정해 줍니다.
최종적으로 리스트의 사이즈를 증가 시켜줍니다.

이전의 정렬 단일연결 리스트와는 다르게 popBackd은 O(1)에 해결 할 수 있으며 정렬 단일 연결 리스트와 마찬가지로 배열기반 리스트와 다르게 인덱스로 접근 할 수 없으므로 이진탐색을 할 수 없어 O(logn)이 아닌 O(n)의 시간복잡도로 실행할 수 밖에 없습니다.
