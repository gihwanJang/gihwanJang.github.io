이진 탐색 트리(Binary Search Tree with parentNode)

앞에서 노드가 값, 왼쪽 자식 노드, 오른쪽 자식 노드를 유지하여 구현한 것을 보았습니다.
해당 경우 remove메소드를 구현시 부모노드를 찾아야 한다는 번거로움이 있었습니다.
그래서 이번 post에서는 노드가 추가적으로 부모 노드를 유지하는 이진 탐색 트리를 구현해 보았습니다.

(코드) 

import java.util.Iterator;
import java.util.Queue;
import java.util.Stack;
import java.util.LinkedList;
import java.util.ArrayList;
import java.util.List;

class BST<E extends Comparable<E>> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private Node right=null;
		private Node left=null;
		private Node parent=null;
		private Node(Object value) {
			this.item=value;
		}
	}
	
	private class BSTIterator implements Iterator<E>{
		private  int cursor =0;
		@Override
		public boolean hasNext() {
			return cursor<visitedOrder.size();
		}
		@Override
		public E next() {
			return (E)visitedOrder.get(cursor++).item;
		}
	}
	
	private Node root = null;
	private int size = 0;
	private List<Node> visitedOrder;
	
	private void BFS(Node node){
		if(this.root==null)return;
		Queue<Node> queue = new LinkedList<>();
		queue.add(node);
		while(!queue.isEmpty()) {
			Node curr=queue.poll();
			this.visitedOrder.add(curr);
			if(curr.left!=null)
				queue.add(curr.left);
			if(curr.right!=null)
				queue.add(curr.right);
		}
	}

	private void preorder(Node node){
		if(this.root==null)return;
		this.visitedOrder.add(node);
		if(node.left!=null) preorder(node.left);
		if(node.right!=null) preorder(node.right);
	}
	
	private void inorder(Node node){
		if(this.root==null)return;
		if(node.left!=null) inorder(node.left);
		this.visitedOrder.add(node);
		if(node.right!=null) inorder(node.right);
	}
	
	private void postorder(Node node){
		if(this.root==null)return;
		if(node.left!=null) postorder(node.left);
		if(node.right!=null) postorder(node.right);
		this.visitedOrder.add(node);
	}
	
	private Node findNode(Node node, Object val) {
		if(((E)node.item).compareTo((E)val)==0) return node;
		else {
			Node nextNode=((E)node.item).compareTo((E)val)>0? node.left:node.right;
			return nextNode!=null? findNode(nextNode, val): node;
		}
	}
	
	private int height(Node node) {
		if(node==null)return 0;
		int h=height(node.left)>height(node.right)?height(node.left):height(node.right);
		return 1+h;
	}
	
	private void removeChild(Node child ) {
		Node descendant=child.left!=null? child.left: child.right;
		if(child.parent==null) this.root = descendant;
		else {
			if(((E)child.parent.item).compareTo((E)child.item)>=0)child.parent.left=descendant;
			else child.parent.right = descendant;
		}
		--this.size;
	}
	
	private void balanceTree(int lo, int hi) {
		if(lo==hi) this.add(visitedOrder.get(lo).item);
		else if(lo+1==hi) {
			this.add(visitedOrder.get(hi).item);
			this.add(visitedOrder.get(lo).item);
		}
		else {
			int mid=lo+(hi-lo)/2;
			this.add(visitedOrder.get(mid).item);
			this.balanceTree(lo,mid-1);
			this.balanceTree(mid+1,hi);
		}
	}
	
	public BST() {
		this.root=null;
		this.size=0;
	}
	
	public BST(Object... arr) {
		for(Object v:arr) this.add(v);
	}
	
	public boolean isEmpty() {
		return this.size==0?true:false;
	}
	
	public int size() {
		return this.size;
	}
	
	public int height() {
		return height(this.root)-1;
	}
	
	public void clear() {
		this.size=0;
		this.root=null;
	}
	
	public Object get(int index) {
		if(index<0||index>this.size-1)return null;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		return visitedOrder.get(index);
	}
	
	public boolean contains(Object val) {
		if(this.isEmpty())return false;
		if(findNode(this.root,val).item==val)return true;
		return false;
	}
	
	public int indexOf(Object val) {
		if(this.isEmpty())return -1;
		visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		int lo=0;
		int hi=this.size-1;
		while(lo<=hi){
			int mid=lo+(hi-lo)/2;
			if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)==0) return mid;
			else if(((E)this.visitedOrder.get(mid).item).compareTo((E) val)>0){
				if(mid==0)return -1;
				hi=mid-1;
			}
			else lo=mid+1;
		}
		return -1;
	}
	
	public void add(Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
		}
		else {
			Node parent=findNode(this.root, val);
			if(((E)parent.item).compareTo((E)val)==0)return;
			else if(((E)parent.item).compareTo((E)val)>0) parent.left=newNode;
			else parent.right=newNode;
			newNode.parent=parent;
		}
		++this.size;
	}
	
	public void remove(Object val) {
		if(this.isEmpty()) return;
		Node delNode = findNode(this.root, val);
		if(((E)delNode.item).compareTo((E)val)!=0) return;
		
		if(delNode.left!=null&&delNode.right!=null) {
			Node prevNode=delNode.left;
			while(prevNode.right!=null)
				prevNode = prevNode.right;
			delNode.item=prevNode.item;
			removeChild(prevNode);
		}
		else removeChild(delNode);
	}
	
	public void balanceTree() {
		this.visitedOrder=new ArrayList<>(this.size);
		inorder(this.root);
		this.clear();
		balanceTree(0, visitedOrder.size()-1);
	}
	
	public BST<E> clone() throws CloneNotSupportedException{
		BST<E> cloned = new BST<E>();
		this.visitedOrder = new ArrayList<>(this.size); 
		this.preorder(this.root);
		for(Node n: visitedOrder)
			cloned.add(n.item);
		return cloned;
	}

	public Iterator<E> iterator(){
		this.visitedOrder= new ArrayList<>(this.size);
		inorder(this.root);
		return new BSTIterator();
	}
	
	@Override
	public String toString() {
		this.visitedOrder=new ArrayList<>(this.size);
		BFS(this.root);
		int childNum=0;
		String str="";
		if(this.isEmpty())return str;
		if(this.visitedOrder.get(0).left!=null)++childNum;
		if(this.visitedOrder.get(0).right!=null)++childNum;
		str+="value : "+this.visitedOrder.get(0).item+", child : "+childNum+"\n";
		for(int i=this.height(),index=1;i>0;--i) {
			int levelsChild=0;
			for(int j=0; j<childNum;++j) {
				int currNodeChild=0;
				if(this.visitedOrder.get(index).left!=null)++currNodeChild;
				if(this.visitedOrder.get(index).right!=null)++currNodeChild;
				str+="value : "+visitedOrder.get(index).item+", child : "+currNodeChild+" ";
				levelsChild+=currNodeChild;
				++index;
			}
			childNum=levelsChild;
			str+="\n";
		}
		return str;
	}
}

(Node 내부 클래스) 

	private static class Node{
		private Object item;
		private Node right=null;
		private Node left=null;
		private Node parent=null;
		private Node(Object value) {
			this.item=value;
		}
	}

이진 탐색 트리의 Node내부 클래스입니다.
해당 노드는 값, 외쪽 자식 노드, 오른쪽 자식 노드 가지는 구성에서 부모 노드를 추가로 가집니다.
parent : 부모 노드를 저장하는 변수로 기본 값이 null로 선언되어 있습니다.

(findNode) 

	private Node findNode(Node node, Object val) {
		if(((E)node.item).compareTo((E)val)==0) return node;
		else {
			Node nextNode=((E)node.item).compareTo((E)val)>0? node.left:node.right;
			return nextNode!=null? findNode(nextNode, val): node;
		}
	}

특정 값의 노드 혹은 해당 값의 부모가 될 노드를 찾는 메소드 입니다.
기존과 동일 하지만 parent를 유지하지 않는 이진 탐색 트리에서는 오버로딩을 통해 findNode메소드가 위의 메소드와 부모노드를 저장하는 스텍을 추가로 받는 메소드 2개가 존재 했지만 parent를 유지하므로 해당 메소드는 필요 없어져 위의 메소드만 남게 됩니다.

(removeChild) 

	private void removeChild(Node child ) {
		Node descendant=child.left!=null? child.left: child.right;
		if(child.parent==null) this.root = descendant;
		else {
			if(((E)child.parent.item).compareTo((E)child.item)>=0)child.parent.left=descendant;
			else child.parent.right = descendant;
		}
		--this.size;
	}

특정 노드와 특정 노드의 부모노드를 받아 특정 노드를 삭제하는 메소드입니다.
parent를 유지하지 않는 이진 탐색 트리에서는 해당 메소드가 매개변수로 부모 노드와 삭제 노드를 받았지만 이제 부모 노드를 유지하고 있음으로 부모 노드를 받지 않아도 됩니다.
그러므로 기존의 매개변수로 받았던 parent를 child.parent로 변경해 줍니다.

(add) 

	public void add(Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
		}
		else {
			Node parent=findNode(this.root, val);
			if(((E)parent.item).compareTo((E)val)==0)return;
			else if(((E)parent.item).compareTo((E)val)>0) parent.left=newNode;
			else parent.right=newNode;
			newNode.parent=parent;
		}
		++this.size;
	}

트리에 새로운 값의 노드를 추가하는 메소드입니다.
기존 메소드에서 추가되는 노드에 부모노드를 지정해주는 과정이 추가되었습니다.

(remove) 

	public void remove(Object val) {
		if(this.isEmpty()) return;
		Node delNode = findNode(this.root, val);
		if(((E)delNode.item).compareTo((E)val)!=0) return;
		
		if(delNode.left!=null&&delNode.right!=null) {
			Node prevNode=delNode.left;
			while(prevNode.right!=null)
				prevNode = prevNode.right;
			delNode.item=prevNode.item;
			removeChild(prevNode);
		}
		else removeChild(delNode);
	}

트리에서 특정 값을 삭제하는 메소드입니다.
기존에 부모노드를 저장하기 위해 선언한 스텍을 제거하였습니다.
또한 부모노드를 찾아 선언한 모든 부분을 제거 하였습니다.

<정리>

구현한 이진 탐색 트리는 삽입하는 순서에 따라 트리의 모습이 결정되기 때문에 삽입 또는 검색시에 앞의 값보다 현재의 값이 계속 커지거나 작아지는 최악의 경우 O(n)의 시간이 걸리게 됩니다.
위는 해당 트리를 완전 이진 트리로 트리를 유지할 시 O(logn)의 시간으로 실행 할 수 있지만 계속해서 balanceTree메소드를 실행하는 것은 비효율적입니다.
이 문제를 해결주는 AVL, RedBlackTree 등이 있습니다.
