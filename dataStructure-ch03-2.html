자료구조 Ch.02 : 트리

<트리>

앞 post에서 본 트리를 간단하게 구현해 보았습니다.
해당 트리는 루트 노드의 레벨을 0으로 보았으며 노드의 순서를 레벨순, 좌측에서 우측 순으로 매겨 구현 하였습니다.
즉 노드의 순서는
1
2 3
과 같이 아래로 갈 수 록 커지고 우측으로 갈 수 록 커지게 순차적으로 생각해 주시면 됩니다.
레벨은 1번 노드의 경우 0이고 2,3번 노드의 경우 1레벨이 됩니다.
그럼 아래에서 주요 메소드 트리의 탐색 삽입 삭제를 위주로 살펴보도록 하겠습니다.

(코드) 

import java.util.Iterator;
import java.util.Queue;
import java.util.Stack;
import java.util.LinkedList;
import java.util.ArrayList;
import java.util.List;

class Tree<E> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private int level;
		private List<Node> childs = new LinkedList<>();
		private Node(Object value) {
			this.item=value;
			this.level=0;
		}
	}
	private Node root = null;
	private int size = 0;
	private int level = 0;

	private List<Node> BFS(Node node){
		List<Node> visitedOrder = new ArrayList<>(this.size);
		if(this.root==null)return visitedOrder;
		Queue<Node> queue = new LinkedList<>();
		queue.add(node);
		while(!queue.isEmpty()) {
			Node curr=queue.poll();
			visitedOrder.add(curr);
			for(Node child: curr.childs) if(child!=null)queue.add(child);
		}
		return visitedOrder;
	}
	
	private List<Node> DFS(Node node){
		List<Node> visitedOrder = new ArrayList<>(this.size);
		if(this.root==null)return visitedOrder;
		Stack<Node> stack = new Stack<>();
		stack.add(node);
		while(!stack.isEmpty()) {
			Node curr=stack.pop();
			visitedOrder.add(curr);
			for(Node child: curr.childs) if(child!=null)stack.add(child);
		}
		return visitedOrder;
	}
	
	private class TreeIterator implements Iterator<E>{
		private  int cursor =0;
		private  List<Node> visitedOrder = BFS(root);
		@Override
		public boolean hasNext() {
			return cursor<visitedOrder.size();
		}
		@Override
		public E next() {
			return (E)visitedOrder.get(cursor++).item;
		}
	}
	
	public Tree(){
		this.root = null;
		this.size = 0;
	}
	
	public Tree(Object[][] arr) {
		for(Object[] l:arr) this.add((int)l[0], l[1]);
	}
	
	public boolean isEmpty() {
		return this.size==0?true:false;
	}
	
	public int size() {
		return this.size;
	}
	
	public void clear() {
		this.root=null;
		this.size=0;
	}
	
	public Object get(int nodeNum) {
		if(nodeNum<1||nodeNum>size)return null;
		List<Node> visitedOrder = BFS(this.root);
		return visitedOrder.get(nodeNum-1);
	}
	
	public void set(int nodeNum, Object val) {
		if(nodeNum<1||nodeNum>size)return;
		List<Node> visitedOrder = BFS(this.root);
		visitedOrder.get(nodeNum-1).item=val;
	}
	
	public boolean contains(Object val) {
		if(this.isEmpty())return false;
		List<Node> visitedOrder = BFS(this.root);
		for(Node n:visitedOrder)if(n.item==val)return true;
		return false;
	}
	
	public int nodeNumOf(Object val) {
		if(this.isEmpty())return -1;
		List<Node> visitedOrder = BFS(this.root);
		for(int i=0; i<visitedOrder.size(); ++i)
			if(visitedOrder.get(i).item==val)return i+1;
		return -1;
	}
	
	public void add(int parentNode, Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
			++this.size;
			return;
		}
		List<Node> visitedOrder = BFS(this.root);
		if(0<parentNode&&parentNode<=visitedOrder.size()) {
			newNode.level=visitedOrder.get(parentNode-1).level+1;
			if(newNode.level>this.level)this.level=newNode.level;
			visitedOrder.get(parentNode-1).childs.add(newNode);
		}
		++this.size;
	}

	public void remove(Object val) {
		List<Node> visitedOrder = DFS(this.root);
		for(int i=0; i<visitedOrder.size(); ++i)
			if(visitedOrder.get(i).item==val)
				for(int j=i-1; j>=0;--j)
					if(visitedOrder.get(j).level+1==visitedOrder.get(i).level) {
						for(Node n:visitedOrder.get(i).childs)
							visitedOrder.get(j).childs.add(n);
						visitedOrder.get(j).childs.remove(visitedOrder.get(i));
            --this.size;
						break;
					}
	}

	public Iterator<E> iterator(){
		return new TreeIterator();
	}
	
	public Tree<E> clone() throws CloneNotSupportedException{
		Tree<E> cloned = new Tree<E>();
		List<Node> visitedOrder = BFS(this.root);
		cloned.add(0,visitedOrder.get(0).item);
		for(int i=0; i<visitedOrder.size(); ++i) {
			for(Node n:visitedOrder.get(i).childs)cloned.add(i+1, n.item);
		}
		return cloned;
	}
	
	@Override
	public String toString() {
		List<Node> visitedOrder = BFS(this.root);
		if(visitedOrder.size()==0)return "null";
		String str="";
		int l=0,p=0,count=0;
		for(int i=0; i<visitedOrder.size();++i){
			if(l!=visitedOrder.get(i).level) {
				str+="\n";
				l=visitedOrder.get(i).level;
			}
			if(count==0) {
				if(i>0&&visitedOrder.get(p).childs.size()==0) {
					str+="(), ";
					++p;
				}
				str+="(";
			}
			str+=visitedOrder.get(i).item;
			if(i==0)str+="), ";
			else if(count==visitedOrder.get(p).childs.size()-1) {
				str+="), ";
				++p;
				count=0;
			}
			else{
				str+=", ";
				++count;
			}
		}
		return str;
	}
}

Tree의 Node내부 클래스입니다.
해당 노드는 값, 노드의 레벨, 자식노드 리스트를 가지고 구성하고 있습니다.
item : 실질적인 데이터를 저장하는 변수입니다.
level : 해당 노드가 위치하는 레벨을 저장하는 변수 입니다.
childs : 해당 노드가 가지는 자식의 리스트로 해당 트리의 경우 자식의 수가 정해 지지 않음으로 LinkedList를 사용하였습니다.

(Tree 멤버변수) 

	private Node root = null;
	private int size = 0;
	private int level = 0;

Tree의 멤버변수입니다.
root : 트리의 최상단 노드를 저장하는 변수로 트리에 1개만이 존재합니다.
size : 노드의 갯수를 저장하는 변수입니다.
level : 트리의 최대 레벨을 저장하는 변수입니다.

(BFS 너비우선 탐색) 

	private List<Node> BFS(Node node){
		List<Node> visitedOrder = new ArrayList<>(this.size);
		if(this.root==null)return visitedOrder;
		Queue<Node> queue = new LinkedList<>();
		queue.add(node);
		while(!queue.isEmpty()) {
			Node curr=queue.poll();
			visitedOrder.add(curr);
			for(Node child: curr.childs) if(child!=null)queue.add(child);
		}
		return visitedOrder;
	}

Tree를 탐색하는 방법중 너비를 우선으로 탑색하는 메소드입니다.
해당 탐색은 너비를 우선으로 탐색을 하므로 각 레벨별로 탐색할 수 있습니다.
탐색한 노드의 정보를 저장할 리스트를 선언해 줍니다.
이때 해당 리스트는 해당 트리의 노드 갯수 만큼 데이터가 나오므로 해당 트리의 size를 용량으로 잡는 ArrayList로 선언해 줍니다.
탐색에 이용할 큐를 선언후 root를 큐에 push해 줍니다.
그리고 반복 적으로 아래의 과정을 반복해 줍니다.

  • 큐에서 노드하나를 꺼냅니다.
  • 꺼낸 노드는 탐색한 노드를 저장하는 리스트에 넣어 줍니다.
  • 꺼낸 노드의 자식이 null이 아니면 모두 큐에 넣우 줍니다.
해당 과정을 반복했을 때 각 레벨 별로 모든 노드를 방문하여 데이터를 받아 올 수 있습니다.

Tree를 탐색하는 방법중 깊이를 우선으로 탑색하는 메소드입니다.
해당 탐색은 깊이를 우선으로 탐색을 하므로 각 간선을 따라 탐색할 수 있습니다.
탐색한 노드의 정보를 저장할 리스트를 선언해 줍니다.
이때 해당 리스트는 해당 트리의 노드 갯수 만큼 데이터가 나오므로 해당 트리의 size를 용량으로 잡는 ArrayList로 선언해 줍니다.
탐색에 이용할 스택 선언후 root를 스택에 push해 줍니다.
그리고 반복 적으로 아래의 과정을 반복해 줍니다.

  • 스택에서 노드하나를 꺼냅니다.
  • 꺼낸 노드는 탐색한 노드를 저장하는 리스트에 넣어 줍니다.
  • 꺼낸 노드의 자식이 null이 아니면 모두 스택에 넣우 줍니다.
해당 과정을 반복했을 때 각 간설을 따라 모든 노드를 방문하여 데이터를 받아 올 수 있습니다.

Tree의 데이터를 순차적으로 접을 할 수 있게 해주는 클래스입니다.
이때 해당 이터레이터는 BFS메소드를 이용하여 리스트를 받아 값을 제공하기 때문에 해당 post에서는 노드의 순서를 레벨순으로 좌측에서 우측 순으로 노드의 순서를 매겼다는 것을 알 수 있습니다.

Tree의 생성자입니다.
인자가 아무것도 없이 생성될 경우 root는 null이고 size는 0인 트리가 생성됩니다.
노드가 있는 트리를 생성해야 할 경우 Object의 2차원 배열을 인자로 받습니다.
해당 배열은 부모노드의 순서와 노드에 들어 갈 데이터 쌍을 가지는 배열입니다.
해당 배열을 받으면 add메소드를 통해 노드를 트리에 넣어 입력받은 노드가 들어간 트리가 생성 됩니다.

(add) 

	public void add(int parentNode, Object val) {
		Node newNode = new Node(val);
		if(this.isEmpty()) {
			this.root=newNode;
			++this.size;
			return;
		}
		List<Node> visitedOrder = BFS(this.root);
		if(0<parentNode&&parentNode<=visitedOrder.size()) {
			newNode.level=visitedOrder.get(parentNode-1).level+1;
			if(newNode.level>this.level)this.level=newNode.level;
			visitedOrder.get(parentNode-1).childs.add(newNode);
      ++this.size;
		}
	}

Tree에 노드를 삽입하는 메소드입니다.
해당 메소드는 매개변수로 부모노드로 설정 할 노드의 순서와 삽입하려는 데이터 두개를 받습니다.
삽입하려는 데이터로 노드를 생성합니다.
이때 만약 트리가 빈트리라면 해당 노드를 루트로 지정해주고 트리의 size를 증가시킨후 종료합니다.
1개라도 노드가 존재한다면 BFS를 통해 노드 리스트를 받아 옵니다.
해당 부모로 설정 할 노드의 번호가 존재한다면 위에서 생성한 노드의 레벨을 부모 노드의 레벨 +1 한 값으로 설정해 줍니다.
이때 설정한 값이 트리의 최대 레벨 보다 높으면 트리의 최대 레벨을 갱신해줍니다.
위의 두 과정이 끝나면 부모노드의 자식으로 생성한 노드를 추가해 주고 트리의 size를 증가시킵니다.

	public void remove(Object val) {
		List<Node> visitedOrder = DFS(this.root);
		for(int i=0; i<visitedOrder.size(); ++i)
			if(visitedOrder.get(i).item==val)
				for(int j=i-1; j>=0;--j)
					if(visitedOrder.get(j).level+1==visitedOrder.get(i).level) {
						for(Node n:visitedOrder.get(i).childs) {
							List<Node> reLevel = DFS(n);
							for(Node r:reLevel)--r.level;
							visitedOrder.get(j).childs.add(n);
						}
						visitedOrder.get(j).childs.remove(visitedOrder.get(i));
						--this.size;
						break;
					}
	}

Tree에 노드를 삭제하는 메소드입니다.
우선 해당 트리에 매개변수로 받은 데이터가 존재하는지 확인 해야하므로 트리를 탐색하여 리스트를 받아 옵니다.
이때 삭제 메소드의 경우 노드의 순서를 고려하는 것이 아닌 삭제 노드의 부모노드를 고려하여기 때문에 DFS를 이용하여 리스트를 받아 왔습니다.
해당 리스트를 선형 검색하여 매개변수로 받은 데이터의 노드가 있다면 해당 노드의 인덱스 좌측에 해당 노드의 부모 노드가 존재하기 때문에 해당 노드의 인덱스-1부터 처음으로 해당 노드 보다 레벨이 1작은 노드를 찾습니다.
부모노드를 찾으면 삭제할 노드의 자식들을 모두 DFS를 통해 아래의 서브트리를 리스트로 받아 해당 노드들을 모드 레벨을 낮춘다음 부모노드에 옮겨 준 후 삭제할 노드를 삭제후 트리의 size를 감소시켜 줍니다.
