import java.util.Iterator;
import java.util.Objects;

class SingleUnsortedLinkedList<E> implements Iterable<E>, Cloneable{
	private static class Node{
		private Object item;
		private Node next=null;
		private Node(Object value) {
			this.item=value;
		}
		public Node clone() throws CloneNotSupportedException{
			Node cloned = (Node)super.clone();
			return cloned;
		}
	}
	private Node head = null;
	private int size = 0;
	
	private class LinkedListIterator implements Iterator<E>{
		private  int cursor =0;
		private Node curr=head;
		@Override
		public boolean hasNext() {
			return cursor<size;
		}
		@Override
		public E next() {
			Object val=curr.item;
			++cursor;
			curr=curr.next;
			return (E)val;
		}
	}
	
	@Override
	public String toString() {
		String str="[";
		Node curr=head;
		for(int i=0;i<size;++i) {
			str+=curr.item;
			curr=curr.next;
			if(i!=this.size-1) {
				str+=", ";
			}
		}
		return str+"]";
	}
	
	public SingleUnsortedLinkedList(){
		this.head = null;
		this.size = 0;
	}
	
	public SingleUnsortedLinkedList(Object... arr){
		for(int i=arr.length-1;i>=0;--i) {
			this.pushFront(arr[i]);
		}
	}
	
	public boolean isEmpty() {
		if(this.size==0)return true;
		return false;
	}
	
	public int size() {
		return this.size;
	}
	
	public void clear() {
		this.size=0;
		this.head=null;
	}
	
	public int indexOf(Object o) {
		if(this.isEmpty())return -1;
		Node curr=head;
		for(int i=0;i<size;++i) {
			if(curr.item==o)return i;
			curr=curr.next;
		}
		return -1;
	}
	
	public boolean contains(Object o) {
		if(this.isEmpty())return false;
		return this.indexOf(o)!=-1?true:false;
	}
	
	public Object get(int index) {
		Objects.checkIndex(index, this.size-1);
		Node curr=this.head;
		for(int i=0;i<index;++i) curr=curr.next;
		return curr.item;
	}
	
	public void pushFront(Object val) {
		Node newNode = new Node(val);
		newNode.next=this.head;
		this.head=newNode;
		++this.size;
	}
	
	public void popFront() {
		if(this.isEmpty())return;
		this.head=head.next;
		--this.size;
	}

	public Object peekFront() {
		return head.item;
	}
	
	public void pushBack(Object val) {
		Node newNode = new Node(val);
		Node curr=this.head;
		for(int i=1; i<size; ++i) curr=curr.next;
		curr.next=newNode;
		++this.size;
	}
	
	public void popBack() {
		if(this.isEmpty())return;
		Node curr=this.head;
		for(int i=2; i<size; ++i) curr=curr.next;
		curr.next=null;
		--this.size;
	}
	
	public Object peekBack() {
		Node curr=head;
		for(int i=1; i<size; ++i) curr=curr.next;
		return curr.item;
	}
	
	public void remove(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr.item!=val) {
			curr=curr.next;
			prev=prev.next;
		}
		if(curr==this.head)this.head=this.head.next;
		else prev.next=curr.next;
		--this.size;
	}
	
	public void removeAll(Object val) {
		if(this.indexOf(val)==-1)return;
		Node curr=this.head;
		Node prev=new Node(null);
		prev.next=this.head;
		while(curr!=null) {
			if(curr.item!=val) {
				curr=curr.next;
				prev=prev.next;
			}
			else {
				if(curr==this.head)this.head=this.head.next;
				else prev.next=curr.next;
				curr=curr.next;
				--this.size;
			}
		}
	}
	
	public Iterator<E> iterator(){
		return new LinkedListIterator();
	}
	
	public SingleUnsortedLinkedList<E> clone() throws CloneNotSupportedException{
		SingleUnsortedLinkedList<E> cloned = (SingleUnsortedLinkedList<E>)super.clone();
		Node curr=cloned.head;
		for(E i:this) {
			Node newNode = new Node(i);
			curr=newNode;
			curr=curr.next;
		}
		return cloned;
	}
}
